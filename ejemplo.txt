    <script>
        document.addEventListener('DOMContentLoaded', () => {
            let currentStep = 1;
            
            // Referencias DOM
            const btnNext = document.getElementById('btn-next');
            const btnBack = document.getElementById('btn-back');
            const btnConfirm = document.getElementById('btn-confirm');
            
            // Inputs Pasos 1
            const cardInput = document.getElementById('cardNumber');
            const cardExpiry = document.getElementById('cardExpiry');
            
            // Formateadores autom√°ticos
            cardInput.addEventListener('input', (e) => {
                let value = e.target.value.replace(/\D/g, '').substring(0, 16);
                let formatted = value.replace(/(.{4})/g, '$1 ').trim();
                e.target.value = formatted;
                clearError('cardNumber');
            });

            cardExpiry.addEventListener('input', (e) => {
                let value = e.target.value.replace(/\D/g, '').substring(0, 4);
                if (value.length >= 2) {
                    value = value.substring(0, 2) + '/' + value.substring(2);
                }
                e.target.value = value;
                clearError('cardExpiry');
            });

            // Limpiar errores gen√©ricos al escribir
            document.querySelectorAll('input').forEach(input => {
                input.addEventListener('input', () => clearError(input.id));
            });

            // --- L√ìGICA DE NAVEGACI√ìN ---
            
            btnNext.addEventListener('click', () => {
                if (validateStep(currentStep)) {
                    changeStep(currentStep + 1);
                }
            });

            btnBack.addEventListener('click', () => {
                if (currentStep > 1) {
                    changeStep(currentStep - 1);
                }
            });

            btnConfirm.addEventListener('click', () => {
                alert('¬°Compra completada con √©xito! Gracias por tu compra.');
                window.location.href = '/'; // O recargar la p√°gina
            });

            function changeStep(step) {
                // Ocultar paso actual
                document.getElementById(`step-content-${currentStep}`).classList.remove('active');
                
                // Actualizar stepper visual (quitar active)
                /* Nota: L√≥gica para mantener "completed" en pasos anteriores */
                updateStepperUI(step);

                currentStep = step;
                
                // Mostrar nuevo paso
                document.getElementById(`step-content-${currentStep}`).classList.add('active');
                
                // Scroll arriba
                window.scrollTo({ top: 0, behavior: 'smooth' });

                // Manejo de botones
                btnBack.style.visibility = currentStep > 1 ? 'visible' : 'hidden';
                
                if (currentStep === 3) {
                    btnNext.style.display = 'none';
                    btnConfirm.style.display = 'block';
                    populateReview(); // Llenar datos de resumen
                } else {
                    btnNext.style.display = 'block';
                    btnConfirm.style.display = 'none';
                }
            }

            function updateStepperUI(newStep) {
                // Resetear todos
                for(let i = 1; i <= 3; i++) {
                    const el = document.getElementById(`step-indicator-${i}`);
                    const circle = el.querySelector('.step-circle');
                    
                    el.classList.remove('active', 'completed');
                    circle.textContent = i; // Resetear n√∫mero
                    
                    if (i < newStep) {
                        el.classList.add('completed');
                        circle.textContent = '‚úì';
                    } else if (i === newStep) {
                        el.classList.add('active');
                    }
                }
            }

            // --- L√ìGICA DE VALIDACI√ìN ---

            function validateStep(step) {
                let isValid = true;
                
                if (step === 1) {
                    // Validar Tarjeta
                    const cNum = document.getElementById('cardNumber').value.replace(/\s/g, '');
                    if (cNum.length < 16) { showError('cardNumber'); isValid = false; }
                    
                    if (!document.getElementById('cardName').value) { showError('cardName'); isValid = false; }
                    
                    const cExp = document.getElementById('cardExpiry').value;
                    if (cExp.length < 5) { showError('cardExpiry'); isValid = false; }
                    
                    const cCvc = document.getElementById('cardCvc').value;
                    if (cCvc.length < 3) { showError('cardCvc'); isValid = false; }
                } 
                else if (step === 2) {
                    // Validar Env√≠o
                    const email = document.getElementById('email').value;
                    const emailRegex = /\S+@\S+\.\S+/;
                    
                    if (!document.getElementById('fullName').value) { showError('fullName'); isValid = false; }
                    if (!emailRegex.test(email)) { showError('email'); isValid = false; }
                    if (!document.getElementById('address').value) { showError('address'); isValid = false; }
                    if (!document.getElementById('city').value) { showError('city'); isValid = false; }
                    if (!document.getElementById('zipCode').value) { showError('zipCode'); isValid = false; }
                }

                return isValid;
            }

            function showError(fieldId) {
                document.getElementById(fieldId).classList.add('error');
                document.getElementById(`error-${fieldId}`).classList.add('visible');
            }

            function clearError(fieldId) {
                document.getElementById(fieldId).classList.remove('error');
                const errorText = document.getElementById(`error-${fieldId}`);
                if (errorText) errorText.classList.remove('visible');
            }

            // --- LLENAR DATOS PASO 3 ---
            function populateReview() {
                // Pago
                const cNum = document.getElementById('cardNumber').value;
                document.getElementById('review-cardNumber').textContent = '**** **** **** ' + cNum.slice(-4);
                document.getElementById('review-cardName').textContent = document.getElementById('cardName').value;

                // Env√≠o
                document.getElementById('review-fullName').textContent = document.getElementById('fullName').value;
                const fullAddress = `${document.getElementById('address').value}, ${document.getElementById('city').value} (CP: ${document.getElementById('zipCode').value})`;
                document.getElementById('review-fullAddress').textContent = fullAddress;
                document.getElementById('review-email').textContent = document.getElementById('email').value;
            }
        });
    </script>
